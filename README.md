## ACTIVATE .venv:
> . .venv/bin/activate
### OR
> source .venv/bin/activate

## IF YOU WANT TO CREATE A NEW VIRTUAL ENVIRONMENT:
> python -m venv .venv

## INSTALL PACKAGES:
> pip install -r requirements.txt

## AFTER EACH PACKAGE INSTALLATION:
> pip freeze > requirements.txt

## RUN THE SCRIPT:
> python MAIN_building_input_df.py
